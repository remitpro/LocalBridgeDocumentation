    SecurityContext API

    Context Structure

    interface SecurityContextType {
    // Legacy compatibility
    isRedactedMode: boolean;
    toggleRedactedMode: () => void;
    clearanceLevel: ClearanceLevel;

    // Enhanced features
    securityState: 'OPEN' | 'ISOLATED' | 'LOCKED';
    currentSession: Session | null;
    auditLog: AuditEvent[];
    metrics: SecurityMetrics;
    isNetworkIsolated: boolean;

    // Actions
    setClearanceLevel: (level: ClearanceLevel) => void;
    logAuditEvent: (type: string, description: string, severity: string) => void;
    startSession: () => void;
    endSession: () => void;
    emergencyLockdown: () => void;
    exportAuditLog: () => string;
    clearAuditLog: () => void;
    getSessionHistory: () => Session[];
}

    Data Types

    type ClearanceLevel = 'UNCLASSIFIED' | 'CONFIDENTIAL' | 'SECRET' | 'TOP_SECRET';

interface AuditEvent {
    id: string;
    timestamp: Date;
    type: string;
    description: string;
    severity: 'low' | 'medium' | 'high' | 'critical';
    userId?: string;
    metadata?: Record<string, any>;
}

interface Session {
    id: string;
    startTime: Date;
    endTime?: Date;
    clearanceLevel: ClearanceLevel;
    events: AuditEvent[];
}

interface SecurityMetrics {
    uptime: number;
    integrityScore: number;
    totalSessions: number;
    activeSessions: number;
    auditEventCount: number;
}

    Usage Example

    import { useSecurity } from '@/context/SecurityContext';

function MyComponent() {
    const {
        isRedactedMode,
        toggleRedactedMode,
        logAuditEvent,
        currentSession
    } = useSecurity();

    const handleAction = () => {
        logAuditEvent('USER_ACTION', 'Button clicked', 'low');
        // ... perform action
    };

    return (
        <div>
            {isRedactedMode && <p>Air-gap active</p>}
            <button onClick={handleAction}>Do Something</button>
        </div>
    );
}

    Encryption Utilities API

    Functions

    deriveKey

    async function deriveKey(password: string, salt: Uint8Array): Promise<CryptoKey>
    Derives a cryptographic key from a password using PBKDF2.

    Parameters: - password: User password - salt: Random salt (16 bytes)

    Returns: CryptoKey for AES-256-GCM

    Algorithm: - PBKDF2 with SHA-256 - 100,000 iterations - 256-bit key length

    encryptText

    async function encryptText(plaintext: string, password: string): Promise<{
    ciphertext: string;
    salt: string;
    iv: string;
}>
    Encrypts text using AES-256-GCM.

    Parameters: - plaintext: Text to encrypt - password: Encryption password

    Returns: Object with base64-encoded ciphertext, salt, and IV

    decryptText

    async function decryptText(
    ciphertext: string,
    password: string,
    salt: string,
    iv: string
): Promise<string>
    Decrypts text encrypted with encryptText.

    Parameters: - ciphertext: Base64-encoded encrypted text - password: Decryption password (must match encryption password) - salt: Base64-encoded salt - iv: Base64-encoded initialization vector

    Returns: Decrypted plaintext

    Throws: Error if password is incorrect or data is corrupted

    encryptFile

    async function encryptFile(file: File, password: string): Promise<{
    encryptedData: Blob;
    salt: string;
    iv: string;
    originalName: string;
    originalSize: number;
}>
    Encrypts a file using AES-256-GCM.

    decryptFile

    async function decryptFile(
    encryptedBlob: Blob,
    password: string,
    salt: string,
    iv: string
): Promise<ArrayBuffer>
    Decrypts a file encrypted with encryptFile.

    calculateHash

    async function calculateHash(data: ArrayBuffer | string): Promise<string>
    Calculates SHA-256 hash of data.

    Returns: Hex-encoded hash string

    calculateFileHash

    async function calculateFileHash(file: File): Promise<string>
    Calculates SHA-256 hash of a file.

    Security Considerations

    1. Password Strength: Minimum 16 characters recommended
    2. Salt: Always use unique random salt
    3. IV: Never reuse IV with same key
    4. Storage: Store salt and IV with ciphertext
    5. Key Derivation: 100,000 iterations provides strong protection

    Network Monitoring Hook

    useNetworkMonitor

    function useNetworkMonitor(isAirGapMode: boolean): {
    networkStatus: {
        isOnline: boolean;
        isIsolated: boolean;
        lastCheck: Date;
    };
    alerts: NetworkAlert[];
    checkIsolation: () => void;
}

    Features: - Real-time network status monitoring - Air-gap violation detection - Automatic alerts when network detected in air-gap mode - Manual isolation verification

    Usage: 

    const { networkStatus, alerts } = useNetworkMonitor(isRedactedMode);

if (alerts.length > 0) {
    // Handle network alerts
}

    File Transfer Flow

    sequenceDiagram
    participant User
    participant Component
    participant Encryption
    participant Context

    User->>Component: Select file
    Component->>Component: Add to queue
    User->>Component: Click upload
    Component->>Encryption: encryptFile(file, password)
    Encryption->>Encryption: Generate salt & IV

    Encryption->>Encryption: Encrypt data
    Encryption-->>Component: {encryptedData, salt, iv}

    Encryption-->>Component: hash
    Component->>Context: logAuditEvent('FILE_TRANSFER', ...)
    Component->>Component: Update status to 'complete'
    Component-->>User: Show download button

    Context-->>Component: Event logged

    Performance Considerations

    Encryption Performance

    • Text Encryption: ~1ms for typical messages
    • File Encryption: Depends on file size
    • 1MB file: ~50-100ms
    • 10MB file: ~500ms-1s
    • 100MB file: ~5-10s

    Memory Usage

    • Audit Log: ~1KB per event
    • Session Data: ~5KB per session
    • File Transfer: File size + encrypted size in memory during transfer

    Optimization Tips

    1. Large Files: Process in chunks for better UX
    2. Audit Log: Limit to last 1000 events
    3. Session History: Archive old sessions
    4. Network Monitoring: Debounce checks to reduce overhead

    Browser Compatibility

    Required APIs

    • Web Crypto API (crypto.subtle)
    • TextEncoder/TextDecoder
    • Blob API
    • File API
    • Local Storage (for future persistence)

    Supported Browsers

    • Chrome 60+
    • Firefox 57+
    • Safari 11+
    • Edge 79+

    Polyfills

    Not required for modern browsers. For older browsers: - Web Crypto API polyfill may be needed - TextEncoder/TextDecoder polyfill

    Testing

    Unit Tests

    describe('encryptionUtils', () => {
    it('should encrypt and decrypt text', async () => {
        const plaintext = 'Secret message';
        const password = 'strong-password-123';

        const encrypted = await encryptText(plaintext, password);
        const decrypted = await decryptText(
            encrypted.ciphertext,
            password,
            encrypted.salt,
            encrypted.iv
        );

        expect(decrypted).toBe(plaintext);
    });

    it('should fail with wrong password', async () => {
        const encrypted = await encryptText('test', 'password1');

        await expect(
            decryptText(encrypted.ciphertext, 'password2', encrypted.salt, encrypted.iv)
        ).rejects.toThrow();
    });
});

    Troubleshooting

    Common Issues

    Encryption Fails

    Cause: Browser doesn't support Web Crypto API
    Solution: Update browser or use polyfill

    Network Alerts in Air-Gap

    Cause: Physical network still connected
    Solution: Disconnect all network interfaces

    Audit Log Not Updating

    Cause: Context not properly initialized
    Solution: Verify SecurityProvider wraps app
