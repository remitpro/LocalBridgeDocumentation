    Fly.io Deployment

    Prerequisites

    • Fly.io account (free tier available)
    • Fly CLI installed
    • PostgreSQL database (Fly Postgres or external)

    Step 1: Install Fly CLI

    # macOS
brew install flyctl

# Linux
curl -L https://fly.io/install.sh | sh

# Windows
pwsh -Command "iwr https://fly.io/install.ps1 -useb | iex"

# Verify installation
flyctl version

    Step 2: Login to Fly.io

    flyctl auth login

    Step 3: Create fly.toml

    Create fly.toml in project root:

    app = "localbridge"
primary_region = "sjc"

[build]
  [build.args]
    NODE_VERSION = "18"

[env]
  PORT = "8080"
  NODE_ENV = "production"

[http_service]
  internal_port = 8080
  force_https = true
  auto_stop_machines = true
  auto_start_machines = true
  min_machines_running = 1
  processes = ["app"]

  [[http_service.checks]]
    interval = "15s"
    timeout = "10s"
    grace_period = "30s"
    method = "GET"
    path = "/health"

[[vm]]
  cpu_kind = "shared"
  cpus = 2
  memory_mb = 4096

[[mounts]]
  source = "localbridge_data"
  destination = "/data"

    Step 4: Create Dockerfile for Fly.io

    # Dockerfile.fly
FROM node:18-alpine AS builder

WORKDIR /app

# Copy package files
COPY package*.json ./
RUN npm ci

# Copy source
COPY . .

# Build
RUN npm run build

# Production image
FROM node:18-alpine

WORKDIR /app

# Copy built files
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/dist-server ./dist-server
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/package.json ./

# Expose port
EXPOSE 8080

# Start
CMD ["npm", "start"]

    Step 5: Set Secrets

    # Set environment variables
flyctl secrets set DATABASE_URL="postgresql://..."
flyctl secrets set JWT_SECRET="your-jwt-secret"
flyctl secrets set SESSION_SECRET="your-session-secret"

# Optional: OAuth secrets
flyctl secrets set GOOGLE_CLIENT_ID="..."
flyctl secrets set GOOGLE_CLIENT_SECRET="..."

    Step 6: Deploy

    # Launch app
flyctl launch

# Deploy
flyctl deploy

# Check status
flyctl status

# View logs
flyctl logs

    Step 7: Set Up Database

    # Create Fly Postgres (optional)
flyctl postgres create

# Attach to app
flyctl postgres attach <postgres-app-name>

# Run migrations
flyctl ssh console
npm run db:migrate
exit

    Step 8: Configure Domain

    # Add custom domain
flyctl certs add your-domain.com

# Get DNS instructions
flyctl certs show your-domain.com

    Docker Deployment

    Step 1: Create Production Dockerfile

    # Dockerfile
FROM node:18-alpine AS builder

WORKDIR /app

# Install dependencies
COPY package*.json ./
RUN npm ci --only=production

# Copy source
COPY . .

# Build
RUN npm run build

# Production image
FROM node:18-alpine

WORKDIR /app

# Install Ollama
RUN apk add --no-cache curl
RUN curl -fsSL https://ollama.ai/install.sh | sh

# Copy built files
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/dist-server ./dist-server
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/package.json ./

# Create data directory
RUN mkdir -p /data

# Expose ports
EXPOSE 3000 11434

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=40s \
  CMD node -e "require('http').get('http://localhost:3000/health', (r) => {process.exit(r.statusCode === 200 ? 0 : 1)})"

# Start script
COPY docker-entrypoint.sh /
RUN chmod +x /docker-entrypoint.sh

ENTRYPOINT ["/docker-entrypoint.sh"]

    Step 2: Create docker-entrypoint.sh

    #!/bin/sh
set -e

# Start Ollama in background
ollama serve &

# Wait for Ollama to be ready
sleep 5

# Pull models if not exists
if ! ollama list | grep -q "llama3.1"; then
  ollama pull llama3.1:latest
fi

# Run migrations
npm run db:migrate

# Start application
exec npm start

    Step 3: Create docker-compose.yml

    version: '3.8'

services:
  app:
    build: .
    ports:
      - "3000:3000"
      - "11434:11434"
    environment:
      - NODE_ENV=production
      - DATABASE_URL=postgresql://postgres:password@db:5432/localbridge
      - JWT_SECRET=${JWT_SECRET}
      - SESSION_SECRET=${SESSION_SECRET}
    depends_on:
      - db
    volumes:
      - ollama_data:/data
    restart: unless-stopped

  db:
    image: postgres:14-alpine
    environment:
      - POSTGRES_DB=localbridge
      - POSTGRES_USER=postgres
      - POSTGRES_PASSWORD=password
    volumes:
      - postgres_data:/var/lib/postgresql/data
    restart: unless-stopped

  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
      - ./ssl:/etc/nginx/ssl:ro
    depends_on:
      - app
    restart: unless-stopped

volumes:
  postgres_data:
  ollama_data:

    Step 4: Deploy with Docker Compose

    # Build and start
docker-compose up -d

# View logs
docker-compose logs -f

# Stop
docker-compose down

# Update
git pull
docker-compose build
docker-compose up -d

    Step 1: Set Up RDS

    # Create PostgreSQL RDS instance
aws rds create-db-instance \
  --db-instance-identifier localbridge-db \
  --db-instance-class db.t3.medium \
  --engine postgres \
  --engine-version 14.7 \
  --master-username admin \
  --master-user-password YourSecurePassword \
  --allocated-storage 20 \
  --vpc-security-group-ids sg-xxxxx \
  --db-subnet-group-name default \
  --backup-retention-period 7 \
  --preferred-backup-window "03:00-04:00" \
  --preferred-maintenance-window "mon:04:00-mon:05:00"

    Step 2: Set Up ECS

    # Create ECS cluster
aws ecs create-cluster --cluster-name localbridge-cluster

# Create task definition
aws ecs register-task-definition --cli-input-json file://task-definition.json

    task-definition.json: 

    {
  "family": "localbridge",
  "networkMode": "awsvpc",
  "requiresCompatibilities": ["FARGATE"],
  "cpu": "2048",
  "memory": "4096",
  "containerDefinitions": [
    {
      "name": "localbridge-api",
      "image": "your-ecr-repo/localbridge:latest",
      "portMappings": [
        {
          "containerPort": 3000,
          "protocol": "tcp"
        }
      ],
      "environment": [
        {
          "name": "NODE_ENV",
          "value": "production"
        }
      ],
      "secrets": [
        {
          "name": "DATABASE_URL",
          "valueFrom": "arn:aws:secretsmanager:..."
        }
      ],
      "logConfiguration": {
        "logDriver": "awslogs",
        "options": {
          "awslogs-group": "/ecs/localbridge",
          "awslogs-region": "us-west-2",
          "awslogs-stream-prefix": "ecs"
        }
      }
    }
  ]
}

    Step 3: Set Up EC2 for Ollama

    # Launch EC2 instance (GPU recommended)
aws ec2 run-instances \
  --image-id ami-xxxxx \
  --instance-type g4dn.xlarge \
  --key-name your-key \
  --security-group-ids sg-xxxxx \
  --subnet-id subnet-xxxxx \
  --user-data file://ollama-setup.sh

    ollama-setup.sh: 

    #!/bin/bash
# Install Ollama
curl -fsSL https://ollama.ai/install.sh | sh

# Pull models
ollama pull llama3.1:latest
ollama pull codellama:latest

# Start Ollama service
systemctl enable ollama
systemctl start ollama

    Step 4: Set Up ALB

    # Create Application Load Balancer
aws elbv2 create-load-balancer \
  --name localbridge-alb \
  --subnets subnet-xxxxx subnet-yyyyy \
  --security-groups sg-xxxxx

# Create target group
aws elbv2 create-target-group \
  --name localbridge-targets \
  --protocol HTTP \
  --port 3000 \
  --vpc-id vpc-xxxxx \
  --health-check-path /health

    Step 5: Deploy Frontend to S3 + CloudFront

    # Build frontend
npm run build

# Upload to S3
aws s3 sync dist/ s3://localbridge-frontend/

# Create CloudFront distribution
aws cloudfront create-distribution \
  --origin-domain-name localbridge-frontend.s3.amazonaws.com \
  --default-root-object index.html

    DigitalOcean Deployment

    Step 1: Create Droplet

    # Using doctl CLI
doctl compute droplet create localbridge \
  --image ubuntu-22-04-x64 \
  --size s-4vcpu-8gb \
  --region sfo3 \
  --ssh-keys your-ssh-key-id

    Step 2: Set Up Server

    # SSH into droplet
ssh root@your-droplet-ip

# Update system
apt update && apt upgrade -y

# Install Node.js
curl -fsSL https://deb.nodesource.com/setup_18.x | bash -
apt install -y nodejs

# Install PostgreSQL
apt install -y postgresql postgresql-contrib

# Install Ollama
curl -fsSL https://ollama.ai/install.sh | sh

# Install Nginx
apt install -y nginx certbot python3-certbot-nginx

    Step 3: Deploy Application

    # Clone repository
git clone https://github.com/your-org/LocalBridge.git
cd LocalBridge

# Install dependencies
npm install

# Build
npm run build

# Set up environment
cp .env.example .env
nano .env  # Edit configuration

# Run migrations
npm run db:migrate

# Install PM2
npm install -g pm2

# Start application
pm2 start npm --name "localbridge" -- start
pm2 save
pm2 startup

    Step 4: Configure Nginx

    # /etc/nginx/sites-available/localbridge
server {
    listen 80;
    server_name your-domain.com;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

    # Enable site
ln -s /etc/nginx/sites-available/localbridge /etc/nginx/sites-enabled/
nginx -t
systemctl reload nginx

# Set up SSL
certbot --nginx -d your-domain.com

    Self-Hosted Deployment

    Hardware Requirements

    Minimum: - Intel i5 or AMD Ryzen 5 - 16GB RAM - 256GB SSD - Gigabit Ethernet

    Recommended: - Intel i7/i9 or AMD Ryzen 7/9 - 32GB+ RAM - 512GB+ NVMe SSD - 10 Gigabit Ethernet - NVIDIA GPU (for faster AI inference)

    Step 1: Prepare Server

    # Install Ubuntu Server 22.04 LTS
# Update system
sudo apt update && sudo apt upgrade -y

# Install required packages
sudo apt install -y \
  git \
  curl \
  build-essential \
  nginx \
  postgresql \
  postgresql-contrib

    Step 2: Install Node.js

    curl -fsSL https://deb.nodesource.com/setup_18.x | sudo -E bash -
sudo apt install -y nodejs

    Step 3: Install Ollama

    curl -fsSL https://ollama.ai/install.sh | sh

# Pull models
ollama pull llama3.1:latest
ollama pull codellama:latest
ollama pull qwen2.5-coder:latest

    Step 4: Set Up Database

    # Create database and user
sudo -u postgres psql
CREATE DATABASE localbridge;
CREATE USER localbridge WITH ENCRYPTED PASSWORD 'your-password';
GRANT ALL PRIVILEGES ON DATABASE localbridge TO localbridge;
\q

    Step 5: Deploy Application

    # Clone repository
cd /opt
sudo git clone https://github.com/your-org/LocalBridge.git
cd LocalBridge

# Install dependencies
sudo npm install

# Configure environment
sudo cp .env.example .env
sudo nano .env

# Build
sudo npm run build

# Run migrations
sudo npm run db:migrate

# Install PM2
sudo npm install -g pm2

# Start application
sudo pm2 start npm --name "localbridge" -- start
sudo pm2 save
sudo pm2 startup systemd

    Step 6: Configure Firewall

    # UFW firewall
sudo ufw allow 22/tcp   # SSH
sudo ufw allow 80/tcp   # HTTP
sudo ufw allow 443/tcp  # HTTPS
sudo ufw enable

    Step 7: Set Up Backups

    # Create backup script
sudo nano /usr/local/bin/backup-localbridge.sh

    #!/bin/bash
BACKUP_DIR="/var/backups/localbridge"
DATE=$(date +%Y%m%d_%H%M%S)

mkdir -p $BACKUP_DIR

# Backup database
pg_dump localbridge | gzip > $BACKUP_DIR/db_$DATE.sql.gz

# Backup application data
tar -czf $BACKUP_DIR/data_$DATE.tar.gz /opt/LocalBridge

# Keep only last 30 days
find $BACKUP_DIR -name "*.gz" -mtime +30 -delete

echo "Backup completed: $DATE"

    # Make executable
sudo chmod +x /usr/local/bin/backup-localbridge.sh

# Add to crontab
sudo crontab -e
# Add: 0 2 * * * /usr/local/bin/backup-localbridge.sh
