    Usage Examples

    1. Input Sanitization in Routes

    import { sanitizeString, sanitizeEmail, sanitizeGithubRepo } from '../lib/sanitizer.js';

router.post('/projects', async (req, res) => {
    // Sanitize inputs
    const name = sanitizeString(req.body.name, { maxLength: 100 });
    const description = sanitizeString(req.body.description, { maxLength: 500 });
    const githubRepo = req.body.githubRepo 
        ? sanitizeGithubRepo(req.body.githubRepo)
        : undefined;

    // Use sanitized values
    const project = await db.insert(projects).values({
        name,
        description,
        githubRepo,
    });

    res.json(project);
});

    2. Email Sanitization

    import { sanitizeEmail } from '../lib/sanitizer.js';

router.post('/register', async (req, res) => {
    try {
        const email = sanitizeEmail(req.body.email);
        // email is now validated and normalized
    } catch (error) {
        throw new ValidationError('Invalid email address');
    }
});

    3. Object Sanitization

    import { sanitizeObject } from '../lib/sanitizer.js';

router.post('/api/data', async (req, res) => {
    // Sanitize entire request body
    const sanitized = sanitizeObject(req.body, {
        maxLength: 1000,
        allowHtml: false,
    });

    // All string values are now sanitized
    await processData(sanitized);
});
